ISSUE 1 — VOICE OUTPUT CUTTING OFF MID SENTENCE:

Replace the current speakText function completely with this chunk-based speaking approach:

function speakInChunks(text, lang) {
  window.speechSynthesis.cancel();
  
  // Split by sentence endings including Kannada/Hindi punctuation
  const sentences = text.match(/[^।.!?]+[।.!?]+/g) || [text];
  
  let index = 0;
  
  function speakNext() {
    if (index >= sentences.length) return;
    
    const utterance = new SpeechSynthesisUtterance(sentences[index].trim());
    utterance.lang = lang;
    utterance.rate = 0.85;
    utterance.pitch = 1.0;
    
    const voices = speechSynthesis.getVoices();
    const voice = voices.find(v => v.lang === lang) ||
                  voices.find(v => v.lang.startsWith(lang.split('-')[0])) ||
                  voices[0];
    if (voice) utterance.voice = voice;
    
    utterance.onend = () => {
      index++;
      speakNext();
    };
    
    utterance.onerror = () => {
      index++;
      speakNext();
    };
    
    window.speechSynthesis.speak(utterance);
  }
  
  // Small delay to let voices load properly
  setTimeout(speakNext, 100);
}

Replace every call to speakText(text) or speak(text) with 
speakInChunks(text, selectedLanguage) where selectedLanguage 
is the current language code like "kn-IN" or "hi-IN"
